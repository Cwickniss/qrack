165  bool M(unsigned int qubitIndex) {
166  if (runningNorm != 1.0) NormalizeState();
167 
168  bool result;
169  double prob = rand_distribution(rand_generator);
170  double angle = rand_distribution(rand_generator) * 2.0 * M_PI;
171  double cosine = cos(angle);
172  double sine = sin(angle);
173 
174  bitCapInt qPower = 1 << qubitIndex;
175  double zeroChance = 0;
176  bitCapInt lcv;
177  for (lcv = 0; lcv < maxQPower; lcv++) {
178  if (lcv & qPower == 0) {
179  zeroChance += normSqrd(stateVec + lcv);
180  }
181  }
182 
183  result = (prob >= zeroChance);
184  double nrmlzr = 1.0;
185  runningNorm = 0.0;
186 
187  if (result) {
188  if (zeroChance < 1.0) nrmlzr = sqrt(1.0 - zeroChance);
189  for (lcv = 0; lcv < maxQPower; lcv++) {
190  if ((lcv & qPower) == 0) {
191  stateVec[lcv] = Complex16(0.0, 0.0);
192  }
193  else {
194  stateVec[lcv] = Complex16(
195  cosine * real(stateVec[lcv]) - sine * imag(stateVec[lcv]),
196  sine * real(stateVec[lcv]) + cosine * imag(stateVec[lcv])
197  ) / nrmlzr;
198  runningNorm += normSqrd(stateVec + lcv);
199  }
200  }
201  }
202  else {
203  if (zeroChance > 0.0) nrmlzr = sqrt(zeroChance);
204  for (lcv = 0; lcv < maxQPower; lcv++) {
205  if ((lcv & qPower) == 0) {
206  stateVec[lcv] = Complex16(
207  cosine * real(stateVec[lcv]) - sine * imag(stateVec[lcv]),
208  sine * real(stateVec[lcv]) + cosine * imag(stateVec[lcv])
209  ) / nrmlzr;
210  runningNorm += normSqrd(stateVec + lcv);
211  }
212  else {
213  stateVec[lcv] = Complex16(0.0, 0.0);
214  }
215  }
216  }
217 
218  runningNorm = sqrt(runningNorm);
219  }
220  bool MAll(bitCapInt fullRegister) {
221  bool result;
222  double prob = rand_distribution(rand_generator);
223  double angle = rand_distribution(rand_generator) * 2.0 * M_PI;
224  double cosine = cos(angle);
225  double sine = sin(angle);
226 
227  Complex16 toTest = stateVec[fullRegister];
228  double oneChance = real(toTest) * real(toTest) + imag(toTest) * imag(toTest);
229  result = (prob < oneChance);
230 
231  double nrmlzr;
232  bitCapInt lcv;
233  bitCapInt maxPower = 1 << qubitCount;
234  if (result) {
235  for (lcv = 0; lcv < maxPower; lcv++) {
236  if (lcv == fullRegister) {
237  stateVec[lcv] = Complex16(cosine, sine);
238  }
239  else {
240  stateVec[lcv] = Complex16(0.0, 0.0);
241  }
242  }
243  }
244  else {
245  nrmlzr = sqrt(1.0 - oneChance);
246  for (lcv = 0; lcv < maxPower; lcv++) {
247  if (lcv == fullRegister) {
248  stateVec[lcv] = Complex16(0.0, 0.0);
249  }
250  else {
251  stateVec[lcv] = Complex16(
252  cosine * real(stateVec[lcv]) - sine * imag(stateVec[lcv]),
253  sine * real(stateVec[lcv]) + cosine * imag(stateVec[lcv])
254  ) / nrmlzr;
255  }
256  }
257  }
258 
259  return result;
260  }
261  double Prob(unsigned int qubitIndex) {

263 
264  bitCapInt qPower = 1 << qubitIndex;
265  double oneChance = 0;
266  bitCapInt lcv;
267  for (lcv = 0; lcv < maxQPower; lcv++) {
268  if ((lcv & qPower) == qPower) {
269  oneChance += normSqrd(stateVec + lcv);
270  }
271  }
272 
273  return oneChance;
274  }

532  void Apply2x2(unsigned int qubitIndex, const Complex16* mtrx) {
533  bitCapInt qPower = 1 << qubitIndex;
534  Complex16 qubit[2];
535  //Complex16 b = Complex16(0.0, 0.0);
536  double sqrNorm = 0.0;
